21/// The theatrical DCI-P3 standard.
22///
23/// This standard uses a gamma 2.6 transfer function and a white point of ~6300K that
24/// matches the color of xenon bulbs used in theater projectors
25#[derive(Copy, Clone, Debug, PartialEq, Eq)]
26pub struct DciP3;

168/// A gamma 2.6 transfer function used by some P3 variants
169///
170/// `P3Gamma` will not use any kind of approximation when converting from `T` to
171/// `T`. This involves calls to `powf`, which may make it too slow for certain
172/// applications.
173///
174/// There are some specialized cases where it has been optimized:
175///
176/// * When converting from `u8` to `f32` or `f64`, while converting to linear
177///   space. This uses lookup tables with precomputed values.
178/// * When converting from `f32` or `f64` to `u8`, while converting from linear
179///   space. This uses a fast algorithm that guarantees a maximum error in the
180///   result of less than 0.6 in line with [this DirectX spec](<https://microsoft.github.io/DirectX-Specs/d3d/archive/D3D11_3_FunctionalSpec.htm#FLOATtoSRGB>).
181#[derive(Copy, Clone, Debug, PartialEq, Eq)]
182pub struct P3Gamma;
183
184impl P3Gamma {
185    /// Access the pre-generated lookup table for non-linear `u8` to linear `f32` conversion.
186    pub fn get_u8_to_f32_lut() -> IntoLinearLut<u8, f32, Self, &'static ArrayTable<256>> {
187        IntoLinearLut::from(P3_GAMMA_U8_TO_F32.get_ref())
188    }
189
190    /// Access the pre-generated lookup table for non-linear `u8` to linear `f64` conversion.
191    pub fn get_u8_to_f64_lut() -> IntoLinearLut<u8, f64, Self, &'static ArrayTable<256>> {
192        IntoLinearLut::from(P3_GAMMA_U8_TO_F64.get_ref())
193    }
194
195    /// Access the pre-generated lookup table for linear `f32` to non-linear `u8` conversion.
196    pub fn get_f32_to_u8_lut() -> FromLinearLut<f32, u8, Self, &'static SliceTable> {
197        FromLinearLut::from_table(P3_GAMMA_F32_TO_U8.get_slice())
198    }
199}
200
201impl GetLutBuilder for P3Gamma {
202    fn get_lut_builder() -> GammaLutBuilder {
203        palette_math::gamma::p3_builder()
204    }
205}

207impl<T> IntoLinear<T, T> for P3Gamma
208where
209    T: Real + Powf,
210{
211    #[inline]
212    fn into_linear(encoded: T) -> T {
213        encoded.powf(T::from_f64(2.6))
214    }
215}
216
217impl<T> FromLinear<T, T> for P3Gamma
218where
219    T: Real + Powf,
220{
221    #[inline]
222    fn from_linear(linear: T) -> T {
223        linear.powf(T::from_f64(1.0 / 2.6))
224    }
225}

228    #[inline]
229    fn into_linear(encoded: u8) -> f32 {
230        *P3_GAMMA_U8_TO_F32.lookup(encoded)
231    }
232}
