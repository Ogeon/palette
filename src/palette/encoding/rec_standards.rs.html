25impl<T: Real> Primaries<T> for Rec2020 {

27    // https://www.itu.int/dms_pubrec/itu-r/rec/bt/R-REC-BT.2020-2-201510-I!!PDF-E.pdf
28    fn red() -> Yxy<Any, T> {
29        Yxy::new(T::from_f64(0.708), T::from_f64(0.292), T::from_f64(0.2627))
30    }
31    fn green() -> Yxy<Any, T> {
32        Yxy::new(T::from_f64(0.170), T::from_f64(0.797), T::from_f64(0.6780))
33    }
34    fn blue() -> Yxy<Any, T> {
35        Yxy::new(T::from_f64(0.131), T::from_f64(0.046), T::from_f64(0.0593))
36    }

92/// The opto-electronic transfer function used in standard dynamic range (SDR)
93/// standards by the ITU-R such as [`Rec709`] and [`Rec2020`].
94///
95/// `RecOetf` will not use any kind of approximation when converting from `T` to
96/// `T`. This involves calls to `powf`, which may make it too slow for certain
97/// applications.
98///
99/// There are some specialized cases where it has been optimized:
100///
101/// * When converting from `u8` to `f32` or `f64`, while converting to linear
102///   space. This uses lookup tables with precomputed values.
103/// * When converting from `f32` or `f64` to `u8`, while converting from linear
104///   space. This uses a fast algorithm that guarantees a maximum error in the
105///   result of less than 0.6 in line with [this DirectX spec](<https://microsoft.github.io/DirectX-Specs/d3d/archive/D3D11_3_FunctionalSpec.htm#FLOATtoSRGB>).
106#[derive(Copy, Clone, Debug, PartialEq, Eq)]
107pub struct RecOetf;
108
109impl RecOetf {
110    /// Access the pre-generated lookup table for non-linear `u8` to linear `f32` conversion.
111    pub fn get_u8_to_f32_lut() -> IntoLinearLut<u8, f32, Self, &'static ArrayTable<256>> {
112        IntoLinearLut::from(REC_OETF_U8_TO_F32.get_ref())
113    }
114
115    /// Access the pre-generated lookup table for non-linear `u8` to linear `f64` conversion.
116    pub fn get_u8_to_f64_lut() -> IntoLinearLut<u8, f64, Self, &'static ArrayTable<256>> {
117        IntoLinearLut::from(REC_OETF_U8_TO_F64.get_ref())
118    }
119
120    /// Access the pre-generated lookup table for linear `f32` to non-linear `u8` conversion.
121    pub fn get_f32_to_u8_lut() -> FromLinearLut<f32, u8, Self, &'static SliceTable> {
122        FromLinearLut::from_table(REC_OETF_F32_TO_U8.get_slice())
123    }
124}

132const ALPHA: f64 = 1.09929682680944;
133const BETA: f64 = 0.018053968510807;
134
135impl<T> IntoLinear<T, T> for RecOetf
136where
137    T: Real + Powf + MulAdd + Arithmetics + PartialCmp + Clone,
138    T::Mask: LazySelect<T>,
139{
140    #[inline]
141    fn into_linear(encoded: T) -> T {
142        lazy_select! {
143            if encoded.lt(&T::from_f64(4.5*BETA)) => T::from_f64(1.0 / 4.5) * &encoded,
144            else => encoded.clone().mul_add(T::from_f64(1.0 / ALPHA), T::from_f64(1.0 - 1.0 / ALPHA)).powf(T::from_f64(1.0 / 0.45))
145        }
146    }
147}
148
149impl<T> FromLinear<T, T> for RecOetf
150where
151    T: Real + Powf + MulSub + Arithmetics + PartialCmp + Clone,
152    T::Mask: LazySelect<T>,
153{
154    #[inline]
155    fn from_linear(linear: T) -> T {
156        lazy_select! {
157            if linear.lt(&T::from_f64(BETA)) => T::from_f64(4.5) * &linear,
158            else => linear.clone().powf(T::from_f64(0.45)).mul_sub(T::from_f64(ALPHA), T::from_f64(ALPHA - 1.0))
159        }
160    }
161}

163impl IntoLinear<f32, u8> for RecOetf {
164    #[inline]
165    fn into_linear(encoded: u8) -> f32 {
166        *REC_OETF_U8_TO_F32.lookup(encoded)
167    }
168}
