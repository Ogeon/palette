12/// A 9 element array representing a 3x3 matrix.
13pub type Mat3<T> = [T; 9];
14pub type Vec3<T> = [T; 3];
15
16/// Multiply the 3x3 matrix with an XYZ color.
17#[inline]
18pub fn multiply_3x3_and_vec3<T>(matrix: Mat3<T>, vector: Vec3<T>) -> Vec3<T>
19where
20    T: Arithmetics,
21{
22    // Input Mat3 and Vec3 are destructured to avoid panic paths.
23    let [m0, m1, m2, m3, m4, m5, m6, m7, m8] = matrix;
24    let [x, y, z] = vector;
25
26    let x1 = m0 * &x;
27    let x2 = m1 * &y;
28    let x3 = m2 * &z;
29
30    let y1 = m3 * &x;
31    let y2 = m4 * &y;
32    let y3 = m5 * &z;
33
34    let z1 = m6 * x;
35    let z2 = m7 * y;
36    let z3 = m8 * z;
37
38    [x1 + x2 + x3, y1 + y2 + y3, z1 + z2 + z3]
39}

41/// Multiply two 3x3 matrices.
42#[inline]
43pub fn multiply_3x3<T>(c: Mat3<T>, f: Mat3<T>) -> Mat3<T>
44where
45    T: Arithmetics + Clone,
46{
47    // Input Mat3 are destructured to avoid panic paths
48    let [c0, c1, c2, c3, c4, c5, c6, c7, c8] = c;
49    let [f0, f1, f2, f3, f4, f5, f6, f7, f8] = f;
50
51    let o0 = c0.clone() * &f0 + c1.clone() * &f3 + c2.clone() * &f6;
52    let o1 = c0.clone() * &f1 + c1.clone() * &f4 + c2.clone() * &f7;
53    let o2 = c0 * &f2 + c1 * &f5 + c2 * &f8;
54
55    let o3 = c3.clone() * &f0 + c4.clone() * &f3 + c5.clone() * &f6;
56    let o4 = c3.clone() * &f1 + c4.clone() * &f4 + c5.clone() * &f7;
57    let o5 = c3 * &f2 + c4 * &f5 + c5 * &f8;
58
59    let o6 = c6.clone() * f0 + c7.clone() * f3 + c8.clone() * f6;
60    let o7 = c6.clone() * f1 + c7.clone() * f4 + c8.clone() * f7;
61    let o8 = c6 * f2 + c7 * f5 + c8 * f8;
62
63    [o0, o1, o2, o3, o4, o5, o6, o7, o8]
64}

66/// Invert a 3x3 matrix and panic if matrix is not invertible.
67#[inline]
68pub fn matrix_inverse<T>(a: Mat3<T>) -> Mat3<T>
69where
70    T: Recip + IsValidDivisor<Mask = bool> + Arithmetics + Clone,
71{
72    // This function runs fastest with assert and no destructuring. The `det`'s
73    // location should not be changed until benched that it's faster elsewhere
74    assert!(a.len() > 8);
75
76    let d0 = a[4].clone() * &a[8] - a[5].clone() * &a[7];
77    let d1 = a[3].clone() * &a[8] - a[5].clone() * &a[6];
78    let d2 = a[3].clone() * &a[7] - a[4].clone() * &a[6];
79    let mut det = a[0].clone() * &d0 - a[1].clone() * &d1 + a[2].clone() * &d2;
80    let d3 = a[1].clone() * &a[8] - a[2].clone() * &a[7];
81    let d4 = a[0].clone() * &a[8] - a[2].clone() * &a[6];
82    let d5 = a[0].clone() * &a[7] - a[1].clone() * &a[6];
83    let d6 = a[1].clone() * &a[5] - a[2].clone() * &a[4];
84    let d7 = a[0].clone() * &a[5] - a[2].clone() * &a[3];
85    let d8 = a[0].clone() * &a[4] - a[1].clone() * &a[3];
86
87    if !det.is_valid_divisor() {
88        panic!("The given matrix is not invertible")
89    }
90    det = det.recip();
91
92    [
93        d0 * &det,
94        -d3 * &det,
95        d6 * &det,
96        -d1 * &det,
97        d4 * &det,
98        -d7 * &det,
99        d2 * &det,
100        -d5 * &det,
101        d8 * det,
102    ]
103}

105/// Maps a matrix from one item type to another.
106///
107/// This turned out to be easier for the compiler to optimize than `matrix.map(f)`.
108#[inline(always)]
109pub fn matrix_map<T, U>(matrix: Mat3<T>, mut f: impl FnMut(T) -> U) -> Mat3<U> {
110    let [m1, m2, m3, m4, m5, m6, m7, m8, m9] = matrix;
111    [
112        f(m1),
113        f(m2),
114        f(m3),
115        f(m4),
116        f(m5),
117        f(m6),
118        f(m7),
119        f(m8),
120        f(m9),
121    ]
122}

124/// Generates the Srgb to Xyz transformation matrix for a given white point.
125#[inline]
126pub fn rgb_to_xyz_matrix<S, T>() -> Mat3<T>
127where
128    S: RgbSpace,
129    S::Primaries: Primaries<T>,
130    S::WhitePoint: WhitePoint<T>,
131    T: Recip + IsValidDivisor<Mask = bool> + Arithmetics + Clone + FromScalar<Scalar = T>,
132    Yxy<Any, T>: IntoColorUnclamped<Xyz<Any, T>>,
133{
134    let r = S::Primaries::red().into_color_unclamped();
135    let g = S::Primaries::green().into_color_unclamped();
136    let b = S::Primaries::blue().into_color_unclamped();
137
138    let matrix = mat3_from_primaries(r, g, b);
139
140    let [s_red, s_green, s_blue] = multiply_3x3_and_vec3(
141        matrix_inverse(matrix.clone()),
142        S::WhitePoint::get_xyz().into(),
143    );
144
145    // Destructuring has some performance benefits, don't change unless measured
146    let [t0, t1, t2, t3, t4, t5, t6, t7, t8] = matrix;
147
148    [
149        t0 * &s_red,
150        t1 * &s_green,
151        t2 * &s_blue,
152        t3 * &s_red,
153        t4 * &s_green,
154        t5 * &s_blue,
155        t6 * s_red,
156        t7 * s_green,
157        t8 * s_blue,
158    ]
159}

161#[rustfmt::skip]
162#[inline]
163fn mat3_from_primaries<T>(r: Xyz<Any, T>, g: Xyz<Any, T>, b: Xyz<Any, T>) -> Mat3<T> {
164    [
165        r.x, g.x, b.x,
166        r.y, g.y, b.y,
167        r.z, g.z, b.z,
168    ]
169}
