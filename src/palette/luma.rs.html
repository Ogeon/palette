39/// A white point and a transfer function.
40pub trait LumaStandard {

42    type WhitePoint;
43
44    /// The transfer function for the luminance component.
45    type TransferFn;
46}

48impl<Wp, Tf> LumaStandard for (Wp, Tf) {
