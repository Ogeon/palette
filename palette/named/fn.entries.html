Get an iterator over all SVG/CSS3 names and colors in arbitrary order.

use palette::Srgb;

let red = Srgb::new(255u8, 0, 0);

let red_entry = palette::named::entries().find(|(name, color)| *color == red);
assert_eq!(red_entry, Some(("red", red)));
