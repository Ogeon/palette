Cast from a Vec of colors to a Vec of color components.

use palette::{cast, Srgb};

let colors = vec![Srgb::new(64u8, 139, 10), Srgb::new(93, 18, 214)];
assert_eq!(
    cast::into_component_vec(colors),
    vec![64, 139, 10, 93, 18, 214]
)
