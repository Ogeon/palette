Cast from a mutable array reference to a mutable color type reference.

use palette::{cast, Srgb};

let mut array = [23, 198, 76];
assert_eq!(cast::from_array_mut::<Srgb<u8>>(&mut array),  &mut Srgb::new(23, 198, 76));
